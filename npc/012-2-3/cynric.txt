    mesq l("Welcome to Candor's Bank!");
    next;

    do
    {
        select
            l("I would like to store some items."),
            l("I would like to perform money transactions."),
            menuaction(l("Quit"));

        mes "";

        switch (@menu) {
            case 1:
                mesn;
                mesq l("Sure thing! I'll have your items transported here from Artis before you realize!");

                closeclientdialog();

                close;
            case 2:

                break;
            default:
                closeclientdialog;
                goodbye;
                close;
        }

    } while (true);
    close;

    .distance = 4;
    end;
}
