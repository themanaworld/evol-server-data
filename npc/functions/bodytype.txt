function	script	stringToGender	{
    switch (ord(strtolower(charat(getarg(0, "n"), 0)))) {
    case 109: // m
        return GENDER_MALE;
    case 102: // f
        return GENDER_FEMALE;
    default:
        return GENDER_HIDDEN;
    }
}

function	script	genderToString	{
    .@sex = getarg(0, Gender);

    if (.@sex == 0)
        return l("female");
    else if (.@sex == 1)
        return l("male");
    else
        return l("hidden");
}
